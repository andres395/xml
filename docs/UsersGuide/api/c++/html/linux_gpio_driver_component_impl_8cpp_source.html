311  void LinuxGpioDriverComponentImpl ::
312  intTaskEntry(void * ptr) {
313 
314  FW_ASSERT(ptr);
315  LinuxGpioDriverComponentImpl* compPtr = (LinuxGpioDriverComponentImpl*) ptr;
316 
317  FW_ASSERT(compPtr->m_fd != -1);
318 
319  // start GPIO interrupt
320  NATIVE_INT_TYPE stat;
321  stat = gpio_set_edge(compPtr->m_gpio, "rising");
322  if (-1 == stat) {
323  compPtr->log_WARNING_HI_GP_IntStartError(compPtr->m_gpio);
324  return;
325  }
326 
327  // spin waiting for interrupt
328  while(not compPtr->m_quitThread) {
329  pollfd fdset[1];
330  NATIVE_INT_TYPE nfds = 1;
331  NATIVE_INT_TYPE timeout = 10000; // Timeout of 10 seconds
332 
333  memset((void*)fdset, 0, sizeof(fdset));
334 
335  fdset[0].fd = compPtr->m_fd;
336  fdset[0].events = POLLPRI;
337 
338  stat = poll(fdset, nfds, timeout);
339 
340  /*
341  * According to this link, poll will always have POLLERR set for the sys/class/gpio subsystem
342  * so cant check for it to look for error:
343  * http://stackoverflow.com/questions/27411013/poll-returns-both-pollpri-pollerr
344  */
345  if (stat < 0) {
346  DEBUG_PRINT("stat: %d, revents: 0x%x, POLLERR: 0x%x, POLLIN: 0x%x, POLLPRI: 0x%x\n",
347  stat, fdset[0].revents, POLLERR, POLLIN, POLLPRI); // TODO remove
348  compPtr->log_WARNING_HI_GP_IntWaitError(compPtr->m_gpio);
349  return;
350  }
351 
352  if (stat == 0) {
353  // continue to poll
354  DEBUG_PRINT("Krait timed out waiting for GPIO interrupt\n");
355  continue;
356  }
357 
358  // Asserting that number of fds w/ revents is 1:
359  FW_ASSERT(stat == 1, stat); // TODO should i bother w/ this assert?
360 
361  // TODO what to do if POLLPRI not set?
362 
363  // TODO: if I take out the read then the poll just continually interrupts
364  // Read is only taking 22 usecs each time, so it is not blocking for long
365  if (fdset[0].revents & POLLPRI) {
366 
367  char *buf[MAX_BUF];
368  (void) lseek(fdset[0].fd, 0, SEEK_SET); // Must seek back to the starting
369  (void) read(fdset[0].fd, buf, MAX_BUF);
370  DEBUG_PRINT("\npoll() GPIO interrupt occurred w/ value: %c\n", buf[0]);
371  }
372 
373  // call interrupt ports
374  Svc::TimerVal timerVal;
375  timerVal.take();
376 
377  for (NATIVE_INT_TYPE port = 0; port < compPtr->getNum_intOut_OutputPorts(); port++) {
378  if (compPtr->isConnected_intOut_OutputPort(port)) {
379  compPtr->intOut_out(port,timerVal);
380  }
381  }
382 
383  }
