60  void SocketIpDriverComponentImpl::readTask(void* pointer) {
61  FW_ASSERT(pointer);
62  SocketIpStatus status = SOCK_SUCCESS;
63  SocketIpDriverComponentImpl* self = reinterpret_cast<SocketIpDriverComponentImpl*>(pointer);
64  do {
65  // Open a network connection if it has not already been open
66  if (not self->m_helper.isOpened()
67  and (self->m_helper.open() != SOCK_SUCCESS)
68  and not self->m_stop) {
69  Os::Task::delay(PRE_CONNECTION_RETRY_INTERVAL_MS);
70  }
71 
72  // If the network connection is open, read from it
73  if (self->m_helper.isOpened()) {
74  BYTE* data = reinterpret_cast<U8*>(self->m_buffer.getdata());
75  FW_ASSERT(data);
76  I32 size = 0;
77  status = self->m_helper.recv(data,size);
78  if (status != SOCK_SUCCESS &&
79  status != SOCK_INTERRUPTED_TRY_AGAIN) {
80  self->m_helper.close();
81  } else {
82  // Ignore KEEPALIVE data and send out any other data.
83  if (memcmp(data, KEEPALIVE_CONST,
84  (size > static_cast<I32>(sizeof(KEEPALIVE_CONST)) - 1) ? sizeof(KEEPALIVE_CONST) -1 : size) != 0) {
85  self->m_buffer.setsize(size);
86  self->recv_out(0, self->m_buffer);
87  }
88  }
89  }
90  }
91  // As long as not told to stop, and we are successful interrupted or ordered to retry, keep receiving
92  while(not self->m_stop && (status == SOCK_SUCCESS ||
93  status == SOCK_INTERRUPTED_TRY_AGAIN || RECONNECT_AUTOMATICALLY != 0));
94  }
